    printf("%d", contiene(string, letra));

}

Debe funcionar con el siguiente código main:


int main(){
    int valor_viejo = 50;
    int valor_nuevo = 100;

    int porcentaje_a = calcularPorcentaje(valor_viejo, valor_nuevo);
    descuentoOAumento(porcentaje_a);

    valor_viejo = 50;
    valor_nuevo = 10;

    system("pause");
}
